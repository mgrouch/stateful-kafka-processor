\begin{abstract}
This document specifies the current implementation exactly for records $T$, $S$, $TS$, signed allocation, ordering, deduplication, and emitted mutation events.
\end{abstract}

\maketitle

\section{Symbols and Exact Record Shapes}
Fix partition keys $P$. All processing is per key $p\in P$.

\begin{definition}[Record $T$]
\[
T=(id,p,pidAlt1,pidAlt2,ref,accId,sorId,oarId,tt,tDate,sDate,tCycle,sMode,a\_status,activity,mStatus,cancel,q,q\_{a,total},q\_{a,\Delta,last},q_f,ledgerTime).
\]
Constraints and defaults:
\begin{itemize}[leftmargin=*]
  \item $id,p,ref$ are non-empty text.
  \item $pidAlt1,pidAlt2,accId,sorId,oarId$ are optional but non-blank if present.
  \item $tt\in\{B,S,CS,SS\}$ with default $B$; $tCycle$ defaults to $SD$; $sMode$ defaults to $CN$; $a\_status\in\{NORM,FAIL\}$ defaults to $NORM$; $activity$ defaults to $A01$; $mStatus$ defaults to $U$.
  \item $q\neq 0$ and sign gate by $tt$: $tt\in\{B,CS\}\Rightarrow q>0$, $tt\in\{S,SS\}\Rightarrow q<0$.
  \item $\operatorname{sgn}(q\_{a,total})\in\{0,\operatorname{sgn}(q)\}$, $|q\_{a,total}|\le |q|$.
  \item $\operatorname{sgn}(q\_{a,\Delta,last})\in\{0,\operatorname{sgn}(q)\}$, $|q\_{a,\Delta,last}|\le |q|$.
  \item $a\_status=FAIL\Rightarrow q_f\neq 0$.
\end{itemize}
\end{definition}

\begin{definition}[Record $S$]
\[
S=(id,p,bDate,q,q_{carry},q_a,q_{a,opposite,\Delta},q_{a,opposite,total},rollover,o,dir,sCycle,ledgerTime).
\]
Constraints and defaults:
\begin{itemize}[leftmargin=*]
  \item $id,p$ are non-empty text.
  \item $dir\in\{D,R\}$ with default from $q$ ($q<0\Rightarrow D$, $q>0\Rightarrow R$); $sCycle$ defaults to $SD$.
  \item $q+q_{carry}\neq 0$.
  \item If $q\neq 0$ and $q_{carry}\neq 0$ then $\operatorname{sgn}(q)=\operatorname{sgn}(q_{carry})$.
  \item $dir=D\Rightarrow q<0$, $dir=R\Rightarrow q>0$.
  \item $\operatorname{sgn}(q_a)\in\{0,\operatorname{sgn}(q+q_{carry})\}$ and $|q_a|\le |q+q_{carry}|$.
\end{itemize}
\end{definition}

\begin{definition}[Record $TS$]
\[
TS=(id,p,pidAlt1,pidAlt2,tid,sid,accId,sorId,oarId,tDate,sDate,q,q_{a,\Delta},q_{a,total,after},tt,activity,mStatus,o,cancel).
\]
Constraints and defaults:
\begin{itemize}[leftmargin=*]
  \item $id,p,tid,sid$ are non-empty text.
  \item $pidAlt1,pidAlt2,accId,sorId,oarId$ are optional but non-blank if present.
  \item $tt$ defaults to $B$ when null; $activity$ defaults to $A01$; $mStatus$ defaults to $U$.
  \item $q\neq 0$.
  \item $\operatorname{sgn}(q_{a,\Delta})=\operatorname{sgn}(q)$ and $|q_{a,\Delta}|\le |q|$.
  \item $\operatorname{sgn}(q_{a,total,after})\in\{0,\operatorname{sgn}(q)\}$ and $|q_{a,total,after}|\le |q|$.
\end{itemize}
\end{definition}

\section{Residuals and Sign-Compatible Allocation}
\[
R_T(T)=q-q_{a,total}.
\]
For $S$, define signed supply usage with carry-first consumption:
\[
R_S(S)=R_{carry}(S)+R_{regular}(S),
\]
where usage magnitude $|q_a|$ consumes $|q_{carry}|$ first, then $|q|$.

Sign compatibility:
\[
Compat(x,y)\iff x\neq 0\land y\neq 0\land \operatorname{sgn}(x)=\operatorname{sgn}(y).
\]
Allocation primitive:
\[
Alloc(x,y)=
\begin{cases}
\operatorname{sgn}(x)\cdot \min(|x|,|y|), & Compat(x,y),\\
0, & \text{otherwise}.
\end{cases}
\]

For any nonzero $\delta=Alloc(x,y)$, both sides are updated by $+\delta$ in their allocated counters.

\section{Per-Key State}
For each $p$:
\begin{itemize}[leftmargin=*]
  \item $\mathcal{T}_p$: open $T$ collection.
  \item $\mathcal{S}_p$: open $S$ collection.
  \item $\mathcal{D}$: dedupe map for incoming $T$ keys $(p,ref,cancel)\mapsto firstSeenTs$.
\end{itemize}
Dedupe window is $W=14$ days (milliseconds in implementation).

\section{Transition Steps}
Each input kind is in $\{T,S,TS\}$. Transition is deterministic given source metadata and state.
Default runtime strategy is \texttt{AutoAllocOppositeStrategy}.

\subsection{Incoming $T$}
\begin{enumerate}[leftmargin=*]
  \item Build dedupe key $(p,ref,cancel)$; if seen within $W$, ignore input.
  \item Else emit mutation \texttt{ACCEPTED\_T}.
  \item If $cancel=true$ and an open stored $T$ with same id exists with $q_{a,total}=0$, force-close it, emit one cancel $TS$, and delete that $T$ from state.
  \item Otherwise load all $S\in\mathcal{S}_p$.
  \item Split by ledger/sign compatibility with current $R_T(T_{in})$ (ledger compatibility is $ledgerTime_T\le ledgerTime_S$ when both present).
  \item Order compatible $S$ by $(\neg rollover,id)$ (thus $rollover=true$ first).
  \item Iterate ordered compatible $S_j$:
  \[
  \delta_j=Alloc(R_T(T),R_S(S_j)).
  \]
  If $\delta_j\neq 0$, update
  \[
  q_{a,total}^{(T)}\gets q_{a,total}^{(T)}+\delta_j,\quad
  q_{a,\Delta,last}^{(T)}\gets \delta_j,\quad
  q_a^{(S_j)}\gets q_a^{(S_j)}+\delta_j,
  \]
  and emit
  \[
  TS=(idPrefix\text{-}k,p,pidAlt1,pidAlt2,tid,sid,accId,sorId,oarId,tDate,bDate,q,\delta_j,q_{a,total}^{(T)},tt,o).
  \]
  \item Append incompatible $S$ unchanged.
  \item Persist each updated $S$ as upsert/delete by open/closed residual.
  \item Persist incoming $T$ as upsert/delete by open/closed residual.
\end{enumerate}
Additionally, when a $T$ becomes fully allocated ($q_{a,total}=q$), set $sDate$ to allocation-side $bDate$ when that date is non-null; otherwise keep prior $sDate$.

\subsection{Incoming $S$}
\begin{enumerate}[leftmargin=*]
  \item Emit mutation \texttt{ACCEPTED\_S}. If $q_{carry}\neq 0$, also forward to the dedicated $S$-with-carry sink.
  \item Load all $T\in\mathcal{T}_p$.
  \item Auto-allocate opposite-sign $T$ first over both ordered and untouched candidate pools, subject to ledger-time compatibility ($ledgerTime_T\le ledgerTime_S$ when both present): set $q_{a,total}\gets q$, set $q_{a,\Delta,last}\gets q-q_{a,total}^{old}$, emit $TS$ when delta is non-zero, and accumulate that delta into $S_{in}.q_{a,opposite,\Delta}$ and $S_{in}.q_{a,opposite,total}$.
  \item Split remaining candidates by ledger/sign compatibility with updated residual $R_S(S_{in})$.
  \item Canonically sort compatible $T$ by $id$, then partition into exact buckets:
  \texttt{PARTIAL\_FAIL\_S}, \texttt{PARTIAL\_FAIL\_NON\_S}, \texttt{FULL\_FAIL\_S}, \texttt{FULL\_FAIL\_NON\_S}, \texttt{PARTIAL\_ALLOC\_S}, \texttt{PARTIAL\_ALLOC\_NON\_S}, \texttt{OPEN\_S}, \texttt{OPEN\_OTHER}.
  \item For each bucket, deterministically shuffle only non-RT ($tCycle\neq RT$) entries with seed
  \[
  hash(seed_0,p,id_{in},\texttt{INCOMING\_S},bucketName),
  \]
  then append RT entries in FIFO order by $(ledgerTime\ \text{nulls-last}, id)$.
  \item Process buckets in the exact order listed above. For each selected $T_i$, apply signed update and emit $TS$ when $\delta\neq 0$.
  \item Append incompatible remaining $T$ unchanged, then append already auto-allocated opposite-sign $T$.
  \item Persist each updated $T$ as upsert/delete by open/closed residual.
  \item Persist incoming $S$ as upsert/delete by open/closed residual.
\end{enumerate}
\subsection{Incoming $TS$}
Forward to processed output and emit mutation \texttt{GENERATED\_TS}.

\section{Mutation Events}
Each emitted mutation record carries:
\[
(type,p,T?,S?,TS?,topic,partition,offset,timestamp,ordinal,eventId).
\]
Event id is deterministic:
\[
eventId = topic\text{-}partition\text{-}offset\text{-}ordinal\text{-}type.
\]
Mutation type set:
\begin{itemize}[leftmargin=*]
  \item \texttt{ACCEPTED\_T}
  \item \texttt{ACCEPTED\_S}
  \item \texttt{GENERATED\_TS}
  \item \texttt{UPSERT\_UNPROCESSED\_T}
  \item \texttt{UPSERT\_UNPROCESSED\_S}
  \item \texttt{DELETE\_UNPROCESSED\_T}
  \item \texttt{DELETE\_UNPROCESSED\_S}
\end{itemize}

\section{Dedupe Eviction}
On stream-time punctuation, delete dedupe entries older than
\[
cutoff=streamTime-W.
\]

\section{Invariants}
\begin{proposition}[Per-key consistency]
All rewritten and emitted entities keep the same partition key $p$.
\end{proposition}

\begin{proposition}[Bounded signed allocation]
\[
|q_{a,total}|\le |q|\ (T),\quad |q_a|\le |q+q_{carry}|\ (S),\quad |q_{a,\Delta}|,|q_{a,total,after}|\le |q|\ (TS).
\]
\end{proposition}

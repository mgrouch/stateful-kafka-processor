\begin{abstract}
This document describes the transition system over $T$, $S$, and $TS$.
It defines exact record shapes, deterministic allocation and ordering rules, deduplication behavior, and emitted mutation events.
It also states cancellation transitions explicitly, including both direct cancellation and compensation cancellation paths.
\end{abstract}

\maketitle

\section{Scope, Motivation, and Notation}
This specification is intentionally \emph{behavior-first}: every rule mirrors runtime decisions over $T$, $S$, and $TS$ states and outputs.
The method is structured this way for three practical reasons:
\begin{itemize}[leftmargin=*]
  \item \textbf{Deterministic replay.} Per-key state, deterministic ordering, and seeded shuffling produce reproducible transitions.
  \item \textbf{Auditability.} Accepted inputs map to explicit mutations, making state transitions reconstructible.
  \item \textbf{Explicit correction semantics.} Cancellation is represented as defined transition outcomes, including compensation when required by current state.
\end{itemize}

Fix partition keys $P$. All processing is per key $p\in P$.
Every rule in this document is deterministic for a fixed input event, source metadata, and current state.

\section{Symbols and Exact Record Shapes}

\begin{definition}[Record $T$]
\[
\begin{aligned}
T={}&(id,p,ref,accId,tt,tDate,sDate,tCycle,sMode,a\_status,cancel,q,\\
&\quad q\_{a,total},q\_{a,\Delta,last},q_f,ledgerTime).
\end{aligned}
\]
Constraints and defaults:
\begin{itemize}[leftmargin=*]
  \item $id,p,ref$ are non-empty text.
  \item $accId$ is optional but non-blank if present.
  \item $tt\in\{B,S,CS,SS\}$ with default $B$.
  \item $tCycle$ defaults to $SD$; $sMode$ defaults to $CN$.
  \item $a\_status\in\{NORM,FAIL\}$ defaults to $NORM$.
  \item $q\neq 0$ and sign gate by $tt$: $tt\in\{B,CS\}\Rightarrow q>0$, $tt\in\{S,SS\}\Rightarrow q<0$.
  \item $\operatorname{sgn}(q\_{a,total})\in\{0,\operatorname{sgn}(q)\}$ and $|q\_{a,total}|\le |q|$.
  \item $\operatorname{sgn}(q\_{a,\Delta,last})\in\{0,\operatorname{sgn}(q)\}$ and $|q\_{a,\Delta,last}|\le |q|$.
  \item $a\_status=FAIL\Rightarrow q_f\neq 0$.
\end{itemize}
\end{definition}

\begin{definition}[Record $S$]
\[
\begin{aligned}
S={}&(id,p,bDate,q,q_{carry},q_a,q_{a,opposite,\Delta},\\
&\quad q_{a,opposite,total},rollover,o,dir,sCycle,ledgerTime).
\end{aligned}
\]
Constraints and defaults:
\begin{itemize}[leftmargin=*]
  \item $id,p$ are non-empty text.
  \item $dir\in\{D,R\}$ with default from $q$ ($q<0\Rightarrow D$, $q>0\Rightarrow R$).
  \item $sCycle$ defaults to $SD$.
  \item $q+q_{carry}\neq 0$.
  \item If $q\neq 0$ and $q_{carry}\neq 0$, then $\operatorname{sgn}(q)=\operatorname{sgn}(q_{carry})$.
  \item $dir=D\Rightarrow q<0$, $dir=R\Rightarrow q>0$.
  \item $\operatorname{sgn}(q_a)\in\{0,\operatorname{sgn}(q+q_{carry})\}$ and $|q_a|\le |q+q_{carry}|$.
\end{itemize}
\end{definition}

\begin{definition}[Record $TS$]
\[
\begin{aligned}
TS={}&(id,p,tid,sid,accId,tDate,sDate,q,q_{a,\Delta},\\
&\quad q_{a,total,after},tt,cancel).
\end{aligned}
\]
Constraints and defaults:
\begin{itemize}[leftmargin=*]
  \item $id,p,tid,sid$ are non-empty text.
  \item $accId$ is optional but non-blank if present.
  \item $tt$ defaults to $B$ when null.
  \item $q\neq 0$.
  \item $\operatorname{sgn}(q_{a,\Delta})=\operatorname{sgn}(q)$ and $|q_{a,\Delta}|\le |q|$.
  \item $\operatorname{sgn}(q_{a,total,after})\in\{0,\operatorname{sgn}(q)\}$ and $|q_{a,total,after}|\le |q|$.
\end{itemize}
\end{definition}

\section{Residuals and Signed Allocation}
Residual (unallocated) quantity of a $T$ record:
\[
R_T(T)=q-q_{a,total}.
\]
For $S$, residual uses carry-first semantics:
\[
R_S(S)=R_{carry}(S)+R_{regular}(S),
\]
where allocation magnitude $|q_a|$ consumes $|q_{carry}|$ first, then $|q|$.

Sign-compatibility predicate:
\[
Compat(x,y)\iff x\neq 0\land y\neq 0\land \operatorname{sgn}(x)=\operatorname{sgn}(y).
\]
Allocation primitive:
\[
Alloc(x,y)=
\begin{cases}
\operatorname{sgn}(x)\cdot \min(|x|,|y|), & \text{if } Compat(x,y),\\
0, & \text{otherwise}.
\end{cases}
\]
For any nonzero $\delta=Alloc(x,y)$, both sides move by $+\delta$ in their allocated counters.

\section{Per-Key State}
For each partition key $p$:
\begin{itemize}[leftmargin=*]
  \item $\mathcal{T}_p$: open $T$ records.
  \item $\mathcal{S}_p$: open $S$ records.
  \item $\mathcal{D}$: dedupe map for incoming $T$ keys $(p,ref,cancel)\mapsto firstSeenTs$.
  \item Processed-marker map entry for previously accepted non-cancel $T$: $(p,id,\texttt{processed})\mapsto firstSeenTs$.
\end{itemize}
Dedupe window is $W=14$ days (in milliseconds in code).

\section{High-Level Transition Flow}
Each input kind is in $\{T,S,TS\}$.
The runtime allocation strategy is \texttt{AutoAllocOppositeStrategy}.

\subsection{Incoming $T$ (normal path and cancel path)}
\begin{enumerate}[leftmargin=*]
  \item Build dedupe key $(p,ref,cancel)$.
  \item If already seen within $W$, drop the record.
  \item Otherwise, store dedupe timestamp and emit mutation \texttt{ACCEPTED\_T}.
  \item If $a\_status=FAIL$, also emit to the failed-$T$ sink.
  \item If $cancel=true$, execute cancellation logic (Section~\ref{sec:cancellation}) and stop normal allocation.
  \item If $cancel=false$, store processed marker $(p,id,\texttt{processed})$ and continue with normal allocation:
  \begin{enumerate}[label=\alph*.,leftmargin=*]
    \item Load all $S\in\mathcal{S}_p$.
    \item Keep candidates that are sign-compatible with $R_T(T_{in})$ and ledger-compatible
    ($ledgerTime_T\le ledgerTime_S$ when both are present).
    \item Order compatible $S$ by $(\neg rollover,id)$, i.e., $rollover=true$ first.
    \item For each ordered candidate $S_j$, compute
    \[
      \delta_j=Alloc(R_T(T),R_S(S_j)).
    \]
    If $\delta_j\neq 0$, update both sides and emit one $TS$.
    \item Persist each updated $S$ as UPSERT or DELETE depending on open residual.
    \item Persist incoming $T$ as UPSERT or DELETE depending on open residual.
  \end{enumerate}
\end{enumerate}
When a $T$ becomes fully allocated ($q_{a,total}=q$), set $sDate$ to the allocating $S.bDate$ when that date is non-null; otherwise keep prior $sDate$.

\subsection{Incoming $S$}
\begin{enumerate}[leftmargin=*]
  \item Emit mutation \texttt{ACCEPTED\_S}.
  \item If $q_{carry}\neq 0$, also forward $S$ to dedicated ``$S$ with carry'' sink.
  \item Load all $T\in\mathcal{T}_p$.
  \item Perform opposite-sign auto-allocation first (over ordered and untouched pools), subject to ledger compatibility
  ($ledgerTime_T\le ledgerTime_S$ when both present). For each non-zero delta:
  \begin{itemize}[leftmargin=*]
    \item set candidate $T$ to fully allocated ($q_{a,total}\gets q$),
    \item set $q_{a,\Delta,last}\gets q-q_{a,total}^{old}$,
    \item emit $TS$,
    \item accumulate delta in $S_{in}.q_{a,opposite,\Delta}$ and $S_{in}.q_{a,opposite,total}$.
  \end{itemize}
  \item With updated residual $R_S(S_{in})$, split remaining $T$ by ledger/sign compatibility.
  \item Sort compatible $T$ by $id$, partition into buckets:
  \begin{itemize}[leftmargin=1.5em]
    \item \texttt{PARTIAL\_FAIL\_S}, \texttt{PARTIAL\_FAIL\_NON\_S}
    \item \texttt{FULL\_FAIL\_S}, \texttt{FULL\_FAIL\_NON\_S}
    \item \texttt{PARTIAL\_ALLOC\_S}, \texttt{PARTIAL\_ALLOC\_NON\_S}
    \item \texttt{OPEN\_S}, \texttt{OPEN\_OTHER}
  \end{itemize}
  \item For each bucket, shuffle non-RT ($tCycle\neq RT$) deterministically with seed
  \[
  \begin{aligned}
  &hash(\\
  &\quad seed_0,p,id_{in},\texttt{INCOMING\_S},bucketName\\
  &).
  \end{aligned}
  \]
  then append RT entries FIFO by $(ledgerTime\ \text{nulls-last},id)$.
  \item Process buckets exactly in listed order; emit $TS$ for each non-zero allocation.
  \item Append incompatible untouched $T$, then append already auto-allocated opposite-sign $T$.
  \item Persist updated $T$ and incoming $S$ as UPSERT/DELETE based on open residual.
\end{enumerate}

\subsection{Incoming $TS$}
Forward to processed output and emit mutation \texttt{GENERATED\_TS}.

\section{Cancellation Semantics (Explicit)}\label{sec:cancellation}
Cancellation is triggered only when an incoming $T$ has $cancel=true$.
The processor executes one of three outcomes:

\subsection{Outcome A: Unknown cancel (ignored)}
If there is no open $T$ with the same id \emph{and} no prior non-cancel processed marker for that id,
the cancel event is ignored after \texttt{ACCEPTED\_T} (with warning log).

\subsection{Outcome B: Direct cancel of unallocated $T$}
If a known $T$ exists and $q_{a,total}=0$:
\begin{enumerate}[leftmargin=*]
  \item Mark the known $T$ as canceled and fully allocated in one step ($q_{a,total}\gets q$).
  \item Create a single cancel $TS$ with
  \begin{itemize}[leftmargin=*]
    \item $sid = \texttt{"cancel-"} + tid$,
    \item $cancel=true$,
    \item $q_{a,\Delta}=q-q_{a,total}^{old}$.
  \end{itemize}
  \item Emit that $TS$ as processed output and as \texttt{GENERATED\_TS} mutation.
  \item If that $T$ was present in open state, remove it and emit \texttt{DELETE\_UNPROCESSED\_T}.
\end{enumerate}

\subsection{Outcome C: Compensation cancel of already-allocated $T$}
If a known $T$ exists and $q_{a,total}\neq 0$:
\begin{enumerate}[leftmargin=*]
  \item Build pseudo-opposite $T$:
  \begin{itemize}[leftmargin=*]
    \item id prefix \texttt{cancel-comp-...},
    \item quantity $q=-q_{a,total}^{known}$,
    \item opposite $tt$ value ($B\leftrightarrow S$, $CS\leftrightarrow SS$),
    \item zero allocated totals at creation.
  \end{itemize}
  \item If original open $T$ exists, delete it from open state and emit \texttt{DELETE\_UNPROCESSED\_T} for the canceled version.
  \item Run normal incoming-$T$ allocation using the pseudo-opposite $T$ against open $S$.
  \item Emit all generated compensation $TS$ events and matching \texttt{GENERATED\_TS} mutations.
  \item Persist updated $S$ plus pseudo-opposite $T$ as UPSERT/DELETE by residual.
\end{enumerate}
Intuition: this path ``unwinds'' previously allocated quantity via a synthetic opposite $T$ rather than rewriting prior $TS$ outputs.

\section{Mutation Events}
Each emitted mutation record carries:
\[
\begin{aligned}
&(type,p,T?,S?,TS?,topic,partition,offset,\\
&\quad timestamp,ordinal,eventId).
\end{aligned}
\]
Event id is deterministic:
\[
\begin{aligned}
eventId ={}& topic\text{-}partition\text{-}offset\\
&\text{-}ordinal\text{-}type.
\end{aligned}
\]
Mutation type set:
\begin{itemize}[leftmargin=*]
  \item \texttt{ACCEPTED\_T}
  \item \texttt{ACCEPTED\_S}
  \item \texttt{GENERATED\_TS}
  \item \texttt{UPSERT\_UNPROCESSED\_T}
  \item \texttt{UPSERT\_UNPROCESSED\_S}
  \item \texttt{DELETE\_UNPROCESSED\_T}
  \item \texttt{DELETE\_UNPROCESSED\_S}
\end{itemize}

\section{Dedupe Eviction}
On stream-time punctuation, delete dedupe entries older than
\[
cutoff=streamTime-W.
\]

\section{Invariants}
\begin{proposition}[Per-key consistency]
All rewritten and emitted entities keep the same partition key $p$.
\end{proposition}

\begin{proposition}[Bounded signed allocation]
\[
\begin{aligned}
&|q_{a,total}|\le |q| \quad (T),\\
&|q_a|\le |q+q_{carry}| \quad (S),\\
&|q_{a,\Delta}|,|q_{a,total,after}|\le |q| \quad (TS).
\end{aligned}
\]
\end{proposition}
